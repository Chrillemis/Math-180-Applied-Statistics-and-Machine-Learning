def linfunc(x, a, b): #Function for linear regression
    return a + b*x

par, cov = curve_fit(linfunc,  Auto["weight"], Auto["mpg"]) #Getting parameters

X = np.linspace(np.min(Auto["weight"]), np.max(Auto["weight"]), 1000)
Y = linfunc(X, *par) #Fit

#plot
